     public DocumentPrinter(Document document) {

     public void printDocument() {
         switch (document.getDocumentType()) {
             case WORD_DOCUMENT:
                 printWORDDocument();
                 break;
             case PDF_DOCUMENT:
                 printPDFDocument();
                 break;

                 printTextDocument();
                 break;

         }
